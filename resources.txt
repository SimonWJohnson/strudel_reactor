# Common Commands
# Check Node and npm versions
node -v
npm -v

# Install dependencies
npm install

# Start local development server
npm start

# Stop running server
Ctrl + C

# Create and switch to dev branch
git checkout -b dev

# Commit and push changes
git add .
git commit -m "Message"
git push origin dev

# Merge dev into main
git checkout main
git merge dev
git push origin main

# Kill port if React server gets stuck
npx kill-port 3000
